<script>

const cards = [

  {img: 'sun.png', name: '太陽', message: '明るい未来が開けます。あなたの努力が実を結ぶ時です。'},

  {img: 'moon.png', name: '月', message: '直感が大きな導きとなります。心の声を信じて。'},

  {img: 'star.png', name: '星', message: '希望が現実に近づいています。素敵な流れです。'},

  {img: 'tower.png', name: '塔', message: '慎重さが求められる時期。焦らないで。'},

  {img: 'heart.png', name: 'ハート', message: '愛や人間関係に嬉しい変化が起こりそう。'},

  {img: 'scythe.png', name: '鎌', message: '終わりと始まり。新しいサイクルの始動です。'}

];


function shuffleCards() {

  const selected = [];

  while (selected.length < 3) {

    const rand = cards[Math.floor(Math.random() * cards.length)];

    if (!selected.includes(rand)) selected.push(rand);

  }


  selected.forEach((card, i) => {

    const cardEl = document.getElementById('card' + (i+1));

    const frontEl = cardEl.querySelector('.front');

    frontEl.src = card.img;


    cardEl.classList.remove('flipped');

    cardEl.onclick = () => {

      cardEl.classList.add('flipped');

      showResult(selected);

    };

  });


  document.getElementById('result').textContent = '';

}


function showResult(selected) {

  const allFlipped = [...document.querySelectorAll('.card')]

    .every(c => c.classList.contains('flipped'));


  if (allFlipped) {

    const messages = selected

      .map(c => `${c.name}：${c.message}`)

      .join(' ／ ');

    document.getElementById('result').textContent = messages;

  }

}


shuffleCards();

</script>
